Loops

Loops are used to run the same code many times.

for i = 0, i < 12, i += 1
  print("i: {i}")
end

let i = 0
while i < 12
  print("Hi!")
  i += 2
end

for value in [2, 4, 6]
  print(value)
end

for id, value in [2, 4, 6]
  print("{id}: {value}")
end

for value in { a: 2, b: 4, c: 6 }
  print(value)
end

for id, value in { a: 2, b: 4, c: 6 }
  print("{id}: {value}")
end
